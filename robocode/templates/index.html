//c++ #include LiquidCrystal.h
const int rs = 13, en = 12, d4 = 11, d5 = 10, d6 = A5, d7 = A4;
LiquidCrystal lcd(rs, en, d4, d5, d6, d7);


int sensorValue;
int sensorPin = A3;
int tc = 100;>


int motor1Pin1 = 2; // pin 2 on IC
int motor1Pin2 = 3; // pin 7 on IC

int motor2Pin1 = 4; // pin 10 on IC
int motor2Pin2 = 5; // pin 15 on IC
// pin 9 on IC>

int state;>
int flag = 0;>
int stateStop = 0;>

void setup() {
lcd.begin(16, 2);

pinMode(sensorPin, INPUT);

// sets the pins as outputs:
pinMode(motor1Pin1, OUTPUT);
pinMode(motor1Pin2, OUTPUT);

pinMode(motor2Pin1, OUTPUT);
pinMode(motor2Pin2, OUTPUT);


// sets enable1Pin and enable2Pin high so that motor can turn on:


// initialize serial communication at 9600 bits per second:
Serial.begin(9600);
}

void loop() {
lcd.setCursor(0, 1);
//Start Serial Communication
if (Serial.available() > 0) {
state = Serial.read();
flag = 0;
}

// if the state is '1' the DC motor will go forward
if (state == '1') {
lcd.print("Forward");
digitalWrite(motor1Pin1, HIGH);
digitalWrite(motor1Pin2, LOW);
digitalWrite(motor2Pin1, HIGH);
digitalWrite(motor2Pin2, LOW);

if (flag == 0) {
Serial.println("Go Forward!");
flag = 1;
}
delay(3000);

}

// if the state is '2' the motor will turn left
else if (state == '2') {
digitalWrite(motor1Pin1, HIGH);
digitalWrite(motor1Pin2, LOW);
digitalWrite(motor2Pin1, LOW);
digitalWrite(motor2Pin2, HIGH);

if (flag == 0) {
Serial.println("Turn LEFT");
flag = 1;
}
delay(3000);

}

// if the state is '3' the motor will Stop
else if (state == '3' || stateStop == 1) {
digitalWrite(motor1Pin1, LOW);
digitalWrite(motor1Pin2, LOW);
digitalWrite(motor2Pin1, LOW);
digitalWrite(motor2Pin2, LOW);
lcd.print("hello, world!");

if (flag == 0) {
Serial.println("STOP!");
flag = 1;
}

}

// if the state is '4' the motor will turn right
else if (state == '4') {
digitalWrite(motor1Pin1, LOW);
digitalWrite(motor1Pin2, HIGH);
digitalWrite(motor2Pin1, HIGH);
digitalWrite(motor2Pin2, LOW);

if (flag == 0) {
Serial.println("Turn RIGHT");
flag = 1;
}

delay(3000);

}

// if the state is '5' the motor will Reverse
else if (state == '5') {
digitalWrite(motor1Pin1, LOW);
digitalWrite(motor1Pin2, HIGH);
digitalWrite(motor2Pin1, LOW);
digitalWrite(motor2Pin2, HIGH);

if (flag == 0) {
Serial.println("Reverse!");
flag = 1;
}

delay(3000);

}
else if (state == '6') {
sensorValue = analogRead(sensorPin); // read analog input pin A5
Serial.print("AirQuality Value: ");
Serial.println(sensorValue, DEC); // prints the value read
}
else if (state == '7') {
digitalWrite(6, HIGH);
Serial.println("Vaccum cleaner");
delay(5000);
}
else if (state == '8') {
digitalWrite(6, LOW);
Serial.println("Vaccum cleaner off");
delay(5000);
}
}
